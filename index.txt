<script>

let display = document.getElementById("display")
let history = document.getElementById("history")

let first = null
let op = null

function press(num){
    display.value += num
}

function operator(o){

    if(first === null){
        first = Number(display.value)
    }else{
        calculate()
        first = Number(display.value)
    }

    op = o
    history.innerText = first + " " + op
    display.value = ""
}

function calculate(){

    let second = Number(display.value)

    history.innerText = first + " " + op + " " + second

    if(op === "+") first = first + second
    if(op === "-") first = first - second
    if(op === "*") first = first * second
    if(op === "/") first = first / second

    display.value = first
}

function clearDisplay(){
    display.value = ""
    history.innerText = ""
    first = null
    op = null
}
